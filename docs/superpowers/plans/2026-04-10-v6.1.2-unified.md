/**
 * Returns true if today is a valid day for evening scans.
 * Evening scans run Sun-Thu to produce picks for the NEXT trading day.
 * Sunday's scan produces Monday's picks using Friday EOD + weekend news.
 */
export function isEveningScanDay(date: Date = new Date()): boolean {
  const day = date.getDay(); // 0=Sun, 1=Mon, ..., 4=Thu
  return day >= 0 && day <= 4;
}

async function callCouncilBrain(opts: CouncilCallOpts): Promise<CouncilMappedResponse> {
  const councilApiUrl = process.env.COUNCIL_API_URL || 'http://localhost:8100';
  const startTime = Date.now();

  let response: Response;
  if (opts.cached) {
    console.log(`[Scan/${opts.trigger}] Fetching cached council output...`);
    response = await fetch(`${councilApiUrl}/latest-output-mapped`, {
      signal: AbortSignal.timeout(60_000),
      next: { revalidate: 0 },
    });
  } else {
    console.log(`[Scan/${opts.trigger}] Triggering live council run...`);
    response = await fetch(`${councilApiUrl}/run-council-mapped?trigger=${opts.trigger}`, {
      method: 'POST',
      signal: AbortSignal.timeout(3_600_000),
      next: { revalidate: 0 },
    });
  }

  if (!response.ok) {
    const body = await response.text().catch(() => 'no body');
    throw new Error(`Council API ${opts.cached ? 'GET' : 'POST'} failed: ${response.status} — ${body}`);
  }

  const data = await response.json() as CouncilMappedResponse;
  console.log(`[Scan/${opts.trigger}] Council response received in ${Date.now() - startTime}ms`);
  return data;
}

async function saveScanReport(opts: SaveReportOpts): Promise<{ reportId: string; spikeCount: number }> {

```typescript
export async function runEveningScan(): Promise<{
  success: boolean;
  picksGenerated: number;
  archiveId?: string;
  deliveredReportId?: string;
  scanDate?: string;
  error?: string;
}> {
  const startTime = Date.now();
  try {
    // 1. Compute tomorrow's date (next trading day)
    const halifaxNow = new Date(new Date().toLocaleString('en-US', { timeZone: 'America/Halifax' }));
    const tomorrow = new Date(halifaxNow);
    tomorrow.setDate(tomorrow.getDate() + 1);
    // Skip to Monday if Saturday/Sunday
    while (tomorrow.getDay() === 0 || tomorrow.getDay() === 6) {
      tomorrow.setDate(tomorrow.getDate() + 1);
    }
    const scanDate = tomorrow.toISOString().split('T')[0];
    const date = new Date(scanDate + 'T12:00:00');
    const scanGeneratedAt = new Date();

    // 2. Call council
    const councilData = await callCouncilBrain({ trigger: 'evening' });

    // 3. Write archive FIRST (immutable, fail-safe)
    const archiveId = await writeScanArchive('EVENING', {
      scanDate,
      scanGeneratedAt,
      runId: councilData.runId || `evening-${scanDate}`,
      regime: councilData.dailyReport?.marketRegime || 'neutral',
      macroContextJson: councilData.macroContext,
      topPicksJson: councilData.spikes,
      councilLogJson: councilData.councilLog,
      riskSummaryJson: councilData.riskSummary,
      dailyRoadmapJson: councilData.dailyRoadmap,
      rawCouncilOutputJson: councilData.rawOutput,
      universeSize: councilData.universeSize || 0,
      tickersScreened: councilData.tickersScreened || 0,
      skippedStagesJson: councilData.skippedStages || [],
    });

    // 4. Save report + spikes + council log
    const { reportId, spikeCount } = await saveScanReport({
      date,
      scanType: 'EVENING',
      councilData,
      spikeData: councilData.spikes || [],
      scanGeneratedAt,
    });

    // 5. Link archive to delivered report
    const model = 'eveningScanArchive';
    await (prisma as any)[model].update({
      where: { id: archiveId },
      data: { deliveredReportId: reportId, deliveredAt: new Date() },
    });

    // 6. Send email
    await sendScanEmail('EVENING', {
      date,
      spikes: councilData.spikes || [],
      councilData,
      isDegraded: (councilData.skippedStages || []).length > 0,
      skippedStages: councilData.skippedStages || [],
      reportId,
    });

    console.log(`[EveningScan] Completed in ${Date.now() - startTime}ms — ${spikeCount} picks`);
    return { success: true, picksGenerated: spikeCount, archiveId, deliveredReportId: reportId, scanDate };
  } catch (error) {
    console.error('[EveningScan] Failed:', error);
    return { success: false, picksGenerated: 0, error: error instanceof Error ? error.message : 'Unknown error' };
  }
}
```

```typescript
import { execSync } from 'child_process';
import { writeFileSync, mkdirSync, existsSync } from 'fs';
import { gzipSync } from 'zlib';
import prisma from '@/lib/db/prisma';

const VAULT_DIR = process.env.VAULT_DIR || '/opt/spike-trades/spiketrades-vault';
const APP_VERSION = '6.1.2';

interface VaultData {
  scanDate: string;
  archiveRow: Record<string, unknown>;
  report: Record<string, unknown>;
  spikes: Record<string, unknown>[];
  councilLog: Record<string, unknown> | null;
}

export async function writeVaultSnapshot(
  scanType: 'EVENING' | 'MORNING',
  data: VaultData
): Promise<void> {
  try {
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    const dateDir = `${VAULT_DIR}/vault/${data.scanDate}`;
    const filename = `${scanType.toLowerCase()}-${timestamp}.json.gz`;
    const filepath = `${dateDir}/${filename}`;

    // Build snapshot
    const snapshot = {
      meta: {
        scanType,
        scanDate: data.scanDate,
        generatedAt: new Date().toISOString(),
        version: APP_VERSION,
      },
      archive: data.archiveRow,
      report: data.report,
      spikes: data.spikes,
      councilLog: data.councilLog,
      accuracyDelta: await getAccuracyDelta(data.scanDate),
      userPortfolioDelta: await getPortfolioDelta(data.scanDate),
    };

    // Compress and write locally
    if (!existsSync(dateDir)) {
      mkdirSync(dateDir, { recursive: true });
    }
    const compressed = gzipSync(JSON.stringify(snapshot));
    writeFileSync(filepath, compressed);
    console.log(`[Vault] Snapshot written: ${filepath} (${compressed.length} bytes)`);

    // Git commit + push (fire-and-forget)
    try {
      execSync(`cd ${VAULT_DIR} && git add vault/ && git commit -m "vault: ${scanType} ${data.scanDate}" && git push`, {
        timeout: 30_000,
        stdio: 'pipe',
      });
      console.log(`[Vault] Pushed to GitHub`);
    } catch (gitErr) {
      console.warn(`[Vault] Git push failed, local copy preserved:`, (gitErr as Error).message);
    }
  } catch (err) {
    console.warn(`[Vault] Snapshot failed:`, (err as Error).message);
  }
}

async function getAccuracyDelta(scanDate: string): Promise<unknown[]> {
  try {
    return await prisma.accuracyRecord.findMany({
      where: { date: new Date(scanDate + 'T12:00:00') },
    });
  } catch {
    return [];
  }
}

async function getPortfolioDelta(scanDate: string): Promise<unknown[]> {
  try {
    const today = new Date(scanDate + 'T00:00:00');
    const tomorrow = new Date(today);
    tomorrow.setDate(tomorrow.getDate() + 1);
    return await prisma.portfolioEntry.findMany({
      where: { createdAt: { gte: today, lt: tomorrow } },
    });
  } catch {
    return [];
  }
}
```

```typescript
import { readFileSync, readdirSync, statSync } from 'fs';
import { gunzipSync } from 'zlib';
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

async function restoreSnapshot(filepath: string) {
  console.log(`Restoring from ${filepath}...`);

  const compressed = readFileSync(filepath);
  const snapshot = JSON.parse(gunzipSync(compressed).toString());

  const { meta, report, spikes, councilLog } = snapshot;
  console.log(`Snapshot: ${meta.scanType} scan for ${meta.scanDate} (v${meta.version})`);

  const dateOnly = new Date(meta.scanDate + 'T12:00:00');
  const scanType = meta.scanType;

  // Upsert DailyReport
  const existingReport = await prisma.dailyReport.findUnique({
    where: { date_scanType: { date: dateOnly, scanType } },
  });

  if (existingReport) {
    // Delete old spikes (scoped WHERE)
    await prisma.spike.deleteMany({ where: { reportId: existingReport.id } });
  }

  const savedReport = await prisma.dailyReport.upsert({
    where: { date_scanType: { date: dateOnly, scanType } },
    create: { date: dateOnly, scanType, ...report },
    update: report,
  });

  // Create spike rows
  for (const spike of spikes) {
    await prisma.spike.create({
      data: { ...spike, reportId: savedReport.id, id: undefined },
    });
  }

  // Upsert CouncilLog
  if (councilLog) {
    await prisma.councilLog.upsert({
      where: { date_scanType: { date: dateOnly, scanType } },
      create: { date: dateOnly, scanType, ...councilLog },
      update: councilLog,
    });
  }

  console.log(`Restored: ${spikes.length} spikes for ${meta.scanDate} ${scanType}`);
}

async function main() {
  const args = process.argv.slice(2);
  if (args.length === 0) {
    console.log('Usage: tsx scripts/restore.ts <snapshot.json.gz> [snapshot2.json.gz] [directory/]');
    process.exit(1);
  }

  for (const arg of args) {
    const stat = statSync(arg);
    if (stat.isDirectory()) {
      // Restore all snapshots in directory
      const files = readdirSync(arg).filter((f) => f.endsWith('.json.gz')).sort();
      for (const file of files) {
        await restoreSnapshot(`${arg}/${file}`);
      }
    } else {
      await restoreSnapshot(arg);
    }
  }

  await prisma.$disconnect();
  console.log('Restore complete.');
}

main().catch(console.error);
```
